**What is a Participant Minute?**
- 1 participant × 1 minute = 1 participant minute
- Example: 50 people on a 60-minute webinar = 3,000 participant minutes
- Cost: 3,000 minutes ÷ 1,000 × $2.40 = $7.20

#### Technical Integration

**Backend Setup:**

```typescript
// lib/100ms.ts
import { SDK } from '@100mslive/server-sdk';

// Initialize SDK
const hmsSDK = new SDK(
  process.env.HMS_APP_ID,
  process.env.HMS_APP_SECRET
);

// Create webinar room
export async function createWebinarRoom(webinar: Webinar) {
  const room = await hmsSDK.rooms.create({
    name: `webinar-${webinar.id}`,
    description: webinar.title,
    template_id: process.env.HMS_WEBINAR_TEMPLATE_ID,
    recording: {
      enabled: true,
      upload_info: {
        type: 's3',
        location: 'webinar-recordings',
        prefix: `${webinar.id}/`,
      },
    },
  });

  // Store room details in database
  await supabase
    .from('webinars')
    .update({
      hms_room_id: room.id,
      hms_room_code: room.room_code,
    })
    .eq('id', webinar.id);

  return room;
}

// Generate auth token for user
export async function generateWebinarToken(
  roomId: string,
  userId: string,
  role: 'host' | 'speaker' | 'attendee'
) {
  const token = await hmsSDK.auth.getAuthToken({
    room_id: roomId,
    user_id: userId,
    role: role,
    type: 'app', // or 'management'
  });

  return token;
}

// End webinar and start processing
export async function endWebinar(roomId: string) {
  await hmsSDK.rooms.disable(roomId);
  
  // Webhook will be triggered when recording is ready
}
```

**Frontend Integration (Prebuilt UI):**

```typescript
// app/webinar/[id]/join/page.tsx
'use client';

import { HMSRoomProvider, HMSPrebuilt } from '@100mslive/roomkit-react';
import { useEffect, useState } from 'react';

export default function WebinarJoinPage({ params }) {
  const [authToken, setAuthToken] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function getToken() {
      try {
        const response = await fetch(`/api/webinar/${params.id}/join`);
        const data = await response.json();
        setAuthToken(data.token);
      } catch (error) {
        console.error('Error getting webinar token:', error);
      } finally {
        setLoading(false);
      }
    }

    getToken();
  }, [params.id]);

  if (loading) {
    return <LoadingSpinner message="Preparing webinar..." />;
  }

  if (!authToken) {
    return <ErrorMessage>Unable to join webinar</ErrorMessage>;
  }

  return (
    <div className="h-screen w-screen">
      <HMSRoomProvider>
        <HMSPrebuilt
          authToken={authToken}
          logo="https://producersavenue.com/logo.png"
          options={{
            userName: currentUser.displayName,
            userId: currentUser.id,
            settings: {
              isAudioMuted: true, // Start muted
              isVideoMuted: false,
            },
          }}
          onLeave={() => {
            window.location.href = `/webinar/${params.id}`;
          }}
        />
      </HMSRoomProvider>
    </div>
  );
}
```

import {
  selectPeers,

function CustomWebinarUI() {
  const peers = useHMSStore(selectPeers);
  const localPeer = useHMSStore(selectLocalPeer);

  );

**API Routes:**

```typescript
// app/api/webinar/[id]/join/route.ts
import { createWebinarRoom, generateWebinarToken } from '@/lib/100ms';

export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  // Verify user is registered for webinar
  const { data: registration } = await supabase
    .from('webinar_registrations')
    .select('*')
    .eq('webinar_id', params.id)
    .eq('user_id', currentUser.id)
    .single();

  if (!registration) {
    return Response.json({ error: 'Not registered' }, { status: 403 });
  }

  // Get or create 100ms room
  const { data: webinar } = await supabase
    .from('webinars')
    .select('*')
    .eq('id', params.id)
    .single();

  if (!webinar.hms_room_id) {
    await createWebinarRoom(webinar);
  }

  // Determine user role
  const role = webinar.host_id === currentUser.id ? 'host' : 'attendee';

  // Generate token
  const token = await generateWebinarToken(
    webinar.hms_room_id,
    currentUser.id,
    role
  );

  // Track attendance
  await supabase.from('webinar_participants').upsert({
    webinar_id: params.id,
    user_id: currentUser.id,
    joined_at: new Date(),
  });

  return Response.json({ token, role });
}
```

**Webhook Handler:**

```typescript
// app/api/webhooks/100ms/route.ts
export async function POST(request: Request) {
  const webhook = await request.json();

  // Verify webhook signature (important!)
  const signature = request.headers.get('100ms-signature');
  const isValid = verify100msSignature(webhook, signature);
  
  if (!isValid) {
    return Response.json({ error: 'Invalid signature' }, { status: 401 });
  }

  switch (webhook.type) {
    case 'recording.success':
      await handleRecordingSuccess(webhook.data);
      break;
    
    case 'recording.failed':
      await handleRecordingFailed(webhook.data);
      break;
    
    case 'beam.stopped':
      // RTMP streaming ended
      break;
  }

  return Response.json({ success: true });
}

async function handleRecordingSuccess(data: any) {
  const { room_id, recording_id, recording_url } = data;

  // Update database
  await supabase
    .from('webinars')
    .update({
      recording_url: recording_url,
      recording_status: 'ready',
      hms_recording_id: recording_id,
    })
    .eq('hms_room_id', room_id);

  // Get webinar and notify attendees
  const { data: webinar } = await supabase
    .from('webinars')
    .select('*, webinar_registrations(*)')
    .eq('hms_room_id', room_id)
    .single();

  // Send email notifications
  await sendBulkEmail({
    to: webinar.webinar_registrations.map(r => r.email),
    template: 'webinar_recording_available',
    data: {
      webinarTitle: webinar.title,
      recordingUrl: `${APP_URL}/webinar/${webinar.id}/recording`,
    },
  });

  // Optional: Auto-convert to product
  if (webinar.convert_to_product) {
    await convertWebinarToProduct(webinar);
  }
}
```

-- Add webinar participants tracking
CREATE TABLE webinar_participants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  webinar_id UUID REFERENCES webinars(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  left_at TIMESTAMPTZ,
  duration_seconds INTEGER,
  role TEXT DEFAULT 'attendee', -- 'host' | 'speaker' | 'attendee'
  was_present BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(webinar_id, user_id)
);

CREATE INDEX idx_webinar_participants_webinar_id ON webinar_participants(webinar_id);
CREATE INDEX idx_webinar_participants_user_id ON webinar_participants(user_id);

CREATE TABLE webinar_analytics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  total_polls INTEGER DEFAULT 0,

#### Alternatives Considered

**Daily.co:**
- **Pros:** More mature (2015), excellent reliability, good docs
- **Cons:** Slightly more expensive ($0.0025/min = $7.50/1000 min), less music-optimized
- **Verdict:** Great alternative, but 100ms offers better DX and pricing

**Agora:**
- **Pros:** Cheapest per-minute ($1.49-3.99/1000 min), ultimate customization
- **Cons:** More complex integration (2-3 days), steeper learning curve, add-ons cost extra
- **Verdict:** Better for large enterprises with custom needs

**Twilio:**
- **Pros:** Very mature (2008), enterprise-grade, extensive features
- **Cons:** More expensive ($4/1000 min), complex pricing, more setup required
- **Verdict:** Overkill for our needs, harder to integrate

**Zoom API:**
- **Pros:** Brand recognition, familiar interface
- **Cons:** Expensive ($40+/month per host), limited customization, external redirect, poor integration
- **Verdict:** Not recommended - bad UX and high cost

**Self-built with WebRTC:**
- **Pros:** Full control, no per-minute costs, ultimate customization
- **Cons:** 6+ months development time, $100k+ cost, ongoing maintenance, complex infrastructure
- **Verdict:** Not viable for MVP, consider for future if scaling massively
